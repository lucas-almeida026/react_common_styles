function flexRow(horizontalAlignment: JustifyContent, verticalAlginment: AlignItems){
  return 'display: flex;' +
    'flex-direction: row;' +
    `justify-content: ${horizontalAlignment};` +
    `align-items: ${verticalAlginment};`
}

function flexColumn(horizontalAlignment: AlignItems, verticalAlginment: JustifyContent){
  return 'display: flex;' +
    'flex-direction: column;' +
    `justify-content: ${horizontalAlignment};` +
    `align-items: ${verticalAlginment};`
}

type JustifyContent = 
  'flex-start' |
  'center' |
  'flex-end' |
  'space-evenly' |
  'space-around' |
  'space-between'

type AlignItems =
  'flex-start' |
  'center' |
  'flex-end'
